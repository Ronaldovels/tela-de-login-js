const divContainer = document.querySelector("#Login")

let isClicked = true

let showOrHide = function(){
    if(isClicked){
        
        isClicked = false
        divContainer.style.opacity = "1"
        
    } else {
        
        divContainer.style.opacity = "0"
        isClicked = true
    }
}
